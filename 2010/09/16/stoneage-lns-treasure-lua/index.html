--Init事件返回值为 true 或 false

function JZ_TestNewNpc_Init( _MeIndex )
        if (Char.SetTalkedEvent(nil, "JZ_TestNewNpc_Talked", _MeIndex) < 0) then
                print("JZ_TestNewNpc_Talked 注册事件失败。");
                return false;
        end
        return true;

end
